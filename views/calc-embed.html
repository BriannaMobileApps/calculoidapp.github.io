Embed this calculator

<script src="http://embed.calculoid.com/scripts/combined.min.js"></script>
<div ng-app="calculoid" ng-controller="CalculoidMainCtrl" ng-init="init({calcId:{{calc.id}},apiKey:'{{user.apiKey}}'})" ng-include="load()"></div>
